speed_detection.c
/* ─────────────────────────────────────────
   Human Speed Detection System
   Real-time velocity via dual sensors
   ───────────────────────────────────────── */

#include <stdio.h>
#include <time.h>
#define SENSOR_GAP   1.0f   // metres
#define SAMPLE_HZ    100    // Hz
#define ALERT_THRESH 2.8f   // m/s

typedef struct {
  float     velocity_ms;
  int       alert_flag;
  clock_t   timestamp;
} SpeedReading;

SpeedReading compute_speed(float elapsed_ms) {
  SpeedReading r;
  float t = elapsed_ms / 1000.0f;
  r.velocity_ms  = (t > 0) ?
                    SENSOR_GAP / t : 0.0f;
  r.alert_flag   = (r.velocity_ms >
                    ALERT_THRESH);
  r.timestamp    = clock();
  return r;
}

void display(SpeedReading const *r) {
  printf("Speed: %.2f m/s\n",
         r->velocity_ms);
  if (r->alert_flag)
    printf("[!] HIGH VELOCITY ALERT\n");
}
